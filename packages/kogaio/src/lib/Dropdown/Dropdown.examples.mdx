```jsx
  return (
    <Dropdown
      colors='dropdown-white'

      label='Single value'
      mx='auto'
      onChange={selectOption}
      value={selectedOption}
      width={{ xs: 1 / 2, md: 1 / 3 }}>
      {options.map(option => (
        <Dropdown.Option key={option.id} value={option.label}>
          {option.label}
        </Dropdown.Option>
      ))}
    </Dropdown>
  )
}
```

```jsx
import { useState } from 'react'
import { Option } from '@ivoryio/kogaio/Dropdown'
const DropdownMulti = () => {
  const options = [
    {
      id: 1,
      label: 'Option 1'
    },
    {
      id: 2,
      label: 'Option 2'
    },
    {
      id: 3,
      label: 'Option 3'
    },
    {
      id: 4,
      label: 'Option 4'
    }
  ]
  const [selectedOptions, setSelectedOptions] = useState([])
  const selectOptions = newOption => {
    if (selectedOptions.includes(newOption)) {
      return setSelectedOptions(prevSelected =>
        prevSelected.filter(op => op !== newOption)
      )
    }
    setSelectedOptions(prevSelected => prevSelected.concat(newOption))
  }
  return (
    <Dropdown
      colors='dropdown-white'
      id='example-dropdown-multi'
      label='Multiple values'
      multiple
      mx='auto'
      onChange={selectOptions}
      value={selectedOptions.join(', ')}
      width={{ xs: 1 / 2, md: 1 / 3 }}>
      {options.map(option => (
        <Option key={option.id} value={option.label}>
          {option.label}
        </Option>
      ))}
    </Dropdown>
  )
}
;<DropdownMulti />
```
